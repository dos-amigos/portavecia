        label: Testo

        label: Immagine
        type: files
        query: page.images
        multiple: false
  previews:
    label: Anteprime
    sections:

        label: Immagini piatti e vini
        type: files
        layout: cards
        image:
          ratio: 1/1
          cover: true
    fields:
      featured_dishes:
        label: Piatti in evidenza
        type: structure
        max: 3
        fields:
          name:
            label: Nome
            type: text
          description:
            label: Descrizione
            type: text

            label: Foto
            type: files
            query: page.images

            label: Foto
            type: files

            multiple: false

        label: Titolo

        type: textarea

        label: Immagine

        query: page.images
